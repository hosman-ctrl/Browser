        function performSearch(query) {

            if (!query) return;

           

            // Add to history

            searchHistory.push({

                query,

                engine: currentEngine.name,

                timestamp: new Date().toISOString(),

                incognito: isIncognito

            });

           

            // Construct search URL

            let searchUrl;

            if (query.includes('.') && !query.includes(' ') &&

                (query.startsWith('http://') || query.startsWith('https://') || query.includes('.'))) {

                // It's a URL

                if (!query.startsWith('http')) {

                    query = 'https://' + query;

                }

                searchUrl = query;

                currentUrl.textContent = query;

            } else {

                // It's a search query

                searchUrl = currentEngine.url + encodeURIComponent(query);

                currentUrl.textContent = `${currentEngine.name} search: ${query}`;

            }

        }
